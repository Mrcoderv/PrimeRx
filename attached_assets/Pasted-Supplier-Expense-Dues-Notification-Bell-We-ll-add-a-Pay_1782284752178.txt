Supplier/Expense Dues + Notification Bell
We'll add a Payables section (dues the pharmacy owes to suppliers, vendors, or pending expenses) and show pending dues in a notification dropdown (like a bell icon in the top navbar).
1. New Model (if not exists)
Create or extend in Models/:
C#// Models/Payable.cs
public class Payable
{
    public int Id { get; set; }
    public string SupplierName { get; set; }
    public string InvoiceNo { get; set; }
    public decimal Amount { get; set; }
    public decimal PaidAmount { get; set; } = 0;
    public DateTime DueDate { get; set; }
    public string Status { get; set; } = "Pending"; // Pending, Paid, Partial
    public string Description { get; set; }
    public DateTime CreatedAt { get; set; } = DateTime.UtcNow;
}

3. JavaScript for Real-time Notifications
Add to _Layout.cshtml or a global JS file:
JavaScriptasync function loadNotifications() {
    try {
        const res = await fetch('/Notifications?handler=GetDues');
        const dues = await res.json();
        
        const countEl = document.getElementById('notificationCount');
        const listEl = document.getElementById('duePayablesList');
        
        countEl.textContent = dues.length;
        
        let html = '';
        dues.forEach(due => {
            html += `
                <li class="dropdown-item">
                    <div class="d-flex justify-content-between">
                        <div>
                            <strong>${due.supplierName}</strong><br>
                            <small>${due.invoiceNo} • Due: ${new Date(due.dueDate).toLocaleDateString()}</small>
                        </div>
                        <span class="text-danger fw-bold">Rs. ${due.pendingAmount}</span>
                    </div>
                </li>
            `;
        });
        
        if (dues.length === 0) {
            html = '<li class="dropdown-item text-muted">No pending dues</li>';
        }
        
        listEl.innerHTML = html;
    } catch(e) { console.error(e); }
}

// Load on page load + every 30 seconds
window.onload = () => {
    loadNotifications();
    setInterval(loadNotifications, 30000);
};
4. Backend Handler (Notifications Page or Shared)
Create Pages/Notifications/Index.cshtml.cs or add to existing:
C#public class NotificationsModel : PageModel
{
    private readonly ApplicationDbContext _context;

    public NotificationsModel(ApplicationDbContext context) => _context = context;

    public async Task<JsonResult> OnGetGetDues()
    {
        var dues = await _context.Payables
            .Where(p => p.Status != "Paid" && p.DueDate <= DateTime.UtcNow.AddDays(7))
            .Select(p => new {
                supplierName = p.SupplierName,
                invoiceNo = p.InvoiceNo,
                dueDate = p.DueDate,
                pendingAmount = p.Amount - p.PaidAmount
            })
            .Take(5)
            .ToListAsync();

        return new JsonResult(dues);
    }
}
