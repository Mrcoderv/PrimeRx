JavaScriptlet selectedItems = [];
let currentFocus = -1;
let medicinesCache = [];

const searchInput = document.getElementById('medicineSearchInput');
const dropdown = document.getElementById('suggestionsDropdown');
const itemsBody = document.getElementById('itemsBody');
const itemsJsonInput = document.getElementById('itemsJson');

// Fetch suggestions
async function fetchSuggestions(term) {
    if (!term || term.length < 1) {
        dropdown.style.display = 'none';
        return;
    }
    
    try {
        const response = await fetch(`/Billing?handler=SearchAsync&term=${encodeURIComponent(term)}`);
        const data = await response.json();
        medicinesCache = data;
        renderSuggestions(data);
    } catch (e) {
        console.error(e);
    }
}

function renderSuggestions(items) {
    dropdown.innerHTML = '';
    if (items.length === 0) {
        dropdown.style.display = 'none';
        return;
    }

    items.forEach((item, index) => {
        const div = document.createElement('div');
        div.className = 'list-group-item';
        div.innerHTML = `
            <strong>${item.name}</strong><br>
            <small>${item.genericName ? '(' + item.genericName + ') ' : ''} 
                   Rs.${item.mrp} | Stock: ${item.stockQuantity}</small>
        `;
        div.dataset.index = index;
        
        div.addEventListener('click', () => {
            addMedicineToBill(item);
            searchInput.value = '';
            dropdown.style.display = 'none';
        });
        
        dropdown.appendChild(div);
    });
    
    dropdown.style.display = 'block';
    currentFocus = -1;
}

// Keyboard navigation
searchInput.addEventListener('keydown', (e) => {
    const items = dropdown.querySelectorAll('.list-group-item');
    
    if (e.key === 'ArrowDown') {
        e.preventDefault();
        currentFocus = Math.min(currentFocus + 1, items.length - 1);
        highlightItem(items);
    } else if (e.key === 'ArrowUp') {
        e.preventDefault();
        currentFocus = Math.max(currentFocus - 1, -1);
        highlightItem(items);
    } else if (e.key === 'Enter' && currentFocus >= 0) {
        e.preventDefault();
        const selected = medicinesCache[currentFocus];
        if (selected) addMedicineToBill(selected);
        searchInput.value = '';
        dropdown.style.display = 'none';
    } else if (e.key === 'Escape') {
        dropdown.style.display = 'none';
    }
});

function highlightItem(items) {
    items.forEach((item, idx) => {
        item.classList.toggle('active', idx === currentFocus);
        if (idx === currentFocus) item.scrollIntoView({ block: 'nearest' });
    });
}

// Live typing
let debounceTimer;
searchInput.addEventListener('input', () => {
    clearTimeout(debounceTimer);
    debounceTimer = setTimeout(() => {
        fetchSuggestions(searchInput.value.trim());
    }, 150);
});

// Click outside to close
document.addEventListener('click', (e) => {
    if (!searchInput.contains(e.target) && !dropdown.contains(e.target)) {
        dropdown.style.display = 'none';
    }
});
